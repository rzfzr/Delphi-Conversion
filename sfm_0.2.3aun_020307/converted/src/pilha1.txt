struct  Tpilha1{
                   int px,py,pV,pN;
                   struct  Tpilha1 *Proximo;
           };
 class  CPilha1  {
    private:
         struct  Tpilha1   *topo;    // somente uma variável
   public:
         void CPilha1::Insere(int x,int y,int V,int N);
         struct  Tpilha1 * CPilha1::Remover(void);
         ~CPilha1(void);		// Destrutor
         CPilha1(void);               // Construtor
};
CPilha1::~CPilha1()
{
  struct Tpilha1 *p;
  while (topo != NULL)
  {
   p=topo;
   topo= topo->Proximo;
   delete p;
  }
}
CPilha1::CPilha1()
{
    topo = NULL;
}

void CPilha1::Insere(int x,int y,int V,int N)
{
  struct Tpilha1 *p;
   p = new (struct Tpilha1);
   p->px = x;
   p->py = y;
   p->pV = V;
   p->pN =  N;
   p->Proximo = topo;
   topo = p;
}
struct  Tpilha1 * CPilha1::Remover(void)
{  struct Tpilha1 *p;
  double auxx;
  if (topo!=NULL)
  {
   p = topo;
   topo = p->Proximo;
   return p;
  }
  else return topo;
}
